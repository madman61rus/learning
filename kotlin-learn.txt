																Функции


Объявление функции :


	//Функция принимает два параметра Int и возвращает Int

	fun sum(a: Int, b: Int): Int {
		return a + b
	}

	//Функция из одного выражения и с выведенным воз. значением

	fun sum(a: Int, b: Int) = a + b

	//Функция, возвращающая общее значение Unit

	func printSum(a: Int, b: Int): Unit {
		print(a+b)
	}

	//Функция с не указанным воз. значением

	func printSum(a: Int, b: Int) {
		print(a+b)
	}
